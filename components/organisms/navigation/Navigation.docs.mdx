```tsx
<Container flexbox>
  {isSmall && <Button label="Open Slideout" onClick={openClicked} />}
  <Navigation
    items={[
      { data: 'zero', iconName: 'action_attach', name: 'one' },
      { data: 'child1', hidden: activeIndex > 2, name: 'child a' },
      { data: 'child2', hidden: activeIndex > 2, name: 'child b' },
      { data: 'one', iconName: 'action_add', name: 'two', useBadge: true},
      { data: 'child3', hidden: activeIndex < 3, name: 'child c' },
      { data: 'child4', hidden: activeIndex < 3, name: 'child d' },
    ]}
    activeIndex={activeIndex}
    onClick={navigationClicked}
    navigationType={!isSmall ? 'pane' : 'slideout'}
    isOpen={isOpen}
    onClose={closeClicked}
    disabledItems={[4]}
    classNames={['pt-7']}
  />
</Container>
```
