### Example Code

This is an example of how to use Link as a router based on [React Router Docs](https://reactrouter.com/web/api/Hooks)

```tsx
import { useHistory } from "react-router-dom";

const MyComponent = () => {

  const history = useHistory();

  const handleRoute = (e: React.MouseEvent) => {
    e.preventDefault();
    history.push('/your-path')
  }

  return (
    <div>
        Example Routing
        <Link onClick={handleRoute}>
    </div>
  )
}
```
